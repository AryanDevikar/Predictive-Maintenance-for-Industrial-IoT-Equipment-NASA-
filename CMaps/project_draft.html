prepare_train_data <- function(data, factor = 0) {
  fd_RUL <- data %>%
    group_by(unit_number) %>%
    summarize(max_time = max(time_in_cycles), .groups = 'drop') # Calculate max per unit_number
  
  data <- merge(data, fd_RUL, by = "unit_number", all.x = TRUE)
  data$RUL <- data$max_time - data$time_in_cycles
  data <- subset(data, select = -max_time)
  data <- data[data$time_in_cycles > factor, ]
  
  return(data)
}

truncate_lifecycle <- function(data, unit_num, percentage) {
  unit_data <- data %>% filter(unit_number == unit_num)
  max_cycle <- max(unit_data$time_in_cycles)
  truncated_cycle <- round(max_cycle * percentage)
  
  truncated_data <- unit_data %>% 
    filter(time_in_cycles <= truncated_cycle)
  
  return(truncated_data)
}

scoring_function <- function(actual, predicted) {
  diff <- predicted - actual
  score <- numeric(length(diff))
  
  score[diff < 0] <- exp(-diff[diff < 0]/13) - 1
  
  score[diff >= 0] <- exp(diff[diff >= 0]/10) - 1
  
  return(mean(score))
}
engineer_features <- function(data, window_sizes = c(5, 10, 15)) {
  result <- data
  
  sensor_cols <- setdiff(names(data), c("unit_number", "time_in_cycles", "RUL", "setting_1", "setting_2"))
  
  units <- unique(data$unit_number)
  
  feature_dfs <- list()
  
  for (unit in units) {
    unit_data <- data[data$unit_number == unit, ]
    unit_features <- unit_data
    
    for (sensor in sensor_cols) {
      for (window in window_sizes) {
        if (nrow(unit_data) >= window) {
          col_name <- paste0(sensor, "_rolling_avg_", window)
          unit_features[[col_name]] <- c(rep(NA, window-1), 
                                         stats::filter(unit_data[[sensor]], rep(1/window, window), sides = 1)[(window):length(unit_data[[sensor]])])
        }
      }
      
      col_name <- paste0(sensor, "_diff")
      unit_features[[col_name]] <- c(NA, diff(unit_data[[sensor]]))
      
      col_name <- paste0(sensor, "_pct_change")
      unit_features[[col_name]] <- c(NA, diff(unit_data[[sensor]]) / unit_data[[sensor]][-nrow(unit_data)] * 100)
      
      for (window in window_sizes) {
        if (nrow(unit_data) >= window) {
          col_name <- paste0(sensor, "_rolling_std_", window)
          roll_sd <- sapply(1:(nrow(unit_data) - window + 1), function(i) {
            sd(unit_data[[sensor]][i:(i+window-1)])
          })
          unit_features[[col_name]] <- c(rep(NA, window-1), roll_sd)
        }
      }
    }
    
    unit_features$norm_cycle <- unit_data$time_in_cycles / max(unit_data$time_in_cycles)
    
    feature_dfs[[length(feature_dfs) + 1]] <- unit_features
  }
  
  result <- do.call(rbind, feature_dfs)
  
  result <- result[complete.cases(result), ]
  
  return(result)
}

engineered_data <- engineer_features(train_fd001)
kfold_cv <- function(data, k = 5, seed = 42) {
  set.seed(seed)
  units <- unique(data$unit_number)
  folds <- createFolds(units, k = k, list = TRUE, returnTrain = FALSE)
  
  fold_data <- list()
  for (i in 1:k) {
    test_units <- units[folds[[i]]]
    train_units <- setdiff(units, test_units)
    
    train_data <- data[data$unit_number %in% train_units, ]
    test_data <- data[data$unit_number %in% test_units, ]
    
    fold_data[[i]] <- list(train = train_data, test = test_data)
  }
  
  return(fold_data)
}

prepare_ml_data <- function(data) {
  X <- data %>% select(-c(unit_number, time_in_cycles, RUL))
  y <- data$RUL
  
  X_matrix <- as.matrix(X)
  
  return(list(X = X, X_matrix = X_matrix, y = y))
}

evaluate_model <- function(actual, predicted, model_name) {
  rmse <- sqrt(mean((predicted - actual)^2))
  mae <- mean(abs(predicted - actual))
  score <- scoring_function(actual, predicted)
  
  cat(paste0("\n", model_name, " Performance:\n"))
  cat(paste0("RMSE: ", round(rmse, 4), "\n"))
  cat(paste0("MAE: ", round(mae, 4), "\n"))
  cat(paste0("PHM08 Score: ", round(score, 4), "\n"))
  
  return(data.frame(model = model_name, rmse = rmse, mae = mae, score = score))
}

cross_validate_model <- function(model_fn, folds, model_name) {
  results <- data.frame()
  fold_metrics <- list()
  
  for (i in 1:length(folds)) {
    cat(paste0("\nFold ", i, " for ", model_name, "\n"))
    
    train_data <- prepare_ml_data(folds[[i]]$train)
    test_data <- prepare_ml_data(folds[[i]]$test)
    
    model <- model_fn(if (model_name == "Linear Regression")

        train_data$X_matrix, train_data$y)
    predictions <- predict(model, if (model_name == "Linear Regression") 
      test_data$X 
      else 
        test_data$X_matrix)
    
    fold_metrics[[i]] <- evaluate_model(test_data$y, predictions, paste0(model_name, " Fold ", i))
  }
  
  all_metrics <- do.call(rbind, fold_metrics)
  
  mean_metrics <- all_metrics %>%
    summarize(
      model = first(model),
      mean_rmse = mean(rmse),
      mean_mae = mean(mae),
      mean_score = mean(score)
    )
  
  return(list(fold_metrics = all_metrics, summary = mean_metrics))
}

  lm(y ~ ., data = as.data.frame(X))
}

train_svr <- function(X, y) {
  X_scaled <- scale(X)
  tune_size <- min(500, nrow(X))
  tune_idx <- sample(1:nrow(X), tune_size)
  
  tune_result <- tune(svm, train.x = X_scaled[tune_idx,], train.y = y[tune_idx],
                     ranges = list(epsilon = seq(0.1, 0.5, 0.2), 
                                   cost = c(1, 10)))
  
  svm(X_scaled, y, kernel = "radial", 
      epsilon = tune_result$best.parameters$epsilon,
      cost = tune_result$best.parameters$cost)
}
